\subsection{\texttt{Git}}
\begin{tabular}{@{}ll@{}}
\texttt{commit -m "<message>"} & \\
\texttt{ls-files} & \\
\texttt{pull}     & \\
\texttt{push}     & \\
\texttt{git log --diff-filter=D --summary | grep delete} &  shows files deleted from repository. \\
\texttt{git log --all -- <filepath>} & shows commit when it was deleted. \\
\texttt{git checkout <SHA> -- <filepath>} & restore the file.
\end{tabular}

\section{\texttt{pdf} tools}
\begin{tabular}{@{}ll@{}}
\verb!pdfcrop [options] <input[.pdf]> [output file]! & crops pdf to the size of the image \\
\end{tabular}
Example:\\
\verb!pdfcrop --margins '-10 -10 -10 -10' input.pdf!

\subsection{\texttt{bash}}
\begin{tabular}{@{}ll@{}}
\verb!vec=( 0.9 0.7 1991.0 )! & define an array \\
\end{tabular}

\subsection{\texttt{sed}}
\begin{tabular}{@{}ll@{}}
\verb!sed -i s/regexp\_old/regexp\_new/g <file>! & replace in the place \\
\end{tabular}

\subsection{\texttt{awk}}
\begin{tabular}{@{}ll@{}}
\verb!awk '/hola/\{print \$1\}' <file>! & prints first record in every line with \verb hola \\
\end{tabular}
