\subsection{\texttt{Git}}
\begin{tabular}{@{}ll@{}}
\verb!git commit -m "<message>"! & \\
\verb!git ls-files! & \\
\verb!git log --diff-filter=D --summary | grep delete! &  shows files deleted from repository. \\
\verb!git log --all -- <filepath>! & shows commit when it was deleted. \\
\verb!git checkout <SHA> -- <filepath>! & restore the file. \\
\verb!git checkout -- .! & restore the file starting with the current root directory. \\
\verb!git show HEAD! & shows last commit \\
\verb!git diff HEAD..HEAD{\raise.17ex\hbox{$\scriptstyle\sim$}}2! & shows differences between last and 2 previous commits \\
\verb!git push origin --delete <branch>! & delete branch on remote repository \\
\end{tabular}

\section{\texttt{pdf} tools}
\begin{tabular}{@{}ll@{}}
\verb!pdfcrop --margins '<left> <top> <right> <bottom>' <input[.pdf]> [output file]! & crops pdf to the size of the image \\
\verb!pdfjoin <file_1.pdf> <file_2.pdf> -o <output_file.pdf>! & join pdfs \\
\verb!pdftk A=<file.pdf> cat A<n> output <output.pdf>! & extract page
\end{tabular}

\subsection{\texttt{bash}}
\begin{tabular}{@{}ll@{}}
\verb!vec=( 0.9 0.7 1991.0 )! & define an array \\
\end{tabular}

\subsection{\texttt{sed}}
\begin{tabular}{@{}ll@{}}
\verb!sed -i s/regexp\_old/regexp\_new/g <file>! & replace in the place \\
\verb!sed -i nd <file>! & deletes line \verb n \\
\verb!sed -i -e 's/for(/for (/g' -e 's/while(/while (/g' -e 's/if(/if (/g' -e 's/){/) {/g' <file.c>! & K-R C indexing \\
\end{tabular}

\subsection{\texttt{awk}}
\begin{tabular}{@{}ll@{}}
\verb!awk '/hola/\{print \$1\}' <file>! & prints first record in every line with \verb hola \\
\end{tabular}

\subsection{\texttt{jobs}}
\begin{tabular}{@{}ll@{}}
\verb!for j in `bjobs | awk '{print $1} '` ; do bkill $j ; done! & kills all the jobs
\end{tabular}
